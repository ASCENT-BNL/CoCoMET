 18def rams_calculate_brightness_temp(rams_xarray: xr.Dataset) -> xr.DataArray:
 19    """
 20
 21
 22    Parameters
 23    ----------
 24    rams_xarray : xarray.core.dataset.Dataset
 25        Xarray Dataset containing default RAMS values.
 26
 27    Returns
 28    -------
 29    TB_xarray : xarray.core.dataarray.DataArray
 30        Xarray DataArray containing brightness temperature at each point and time--same dimension as input [K].
 31
 32    """
 33
 34    rams_xarray["TOA_OLR"] = (
 35        ["Time", "south_north", "west_east"],
 36        rams_xarray["LWUP"][:, -1, :, :].values,
 37    )  # the top of atmosphere surface radiation, needed to TB calculation
 38    TOA_OLR = rams_xarray["TOA_OLR"].values
 39
 40    TB = np.zeros(TOA_OLR.shape)
 41
 42    a = 1.228
 43    b = -1.106e-3
 44    sigma = 5.67e-8  # W m^-2 K^-4
 45
 46    print("=====Calculating RAMS Brightness Temperatures=====")
 47    tf = (TOA_OLR[:, :, :] / sigma) ** 0.25
 48    TB[:, :, :] = (np.sqrt(4 * b * tf + a**2) - a) / (2 * b)
 49
 50    TB_xarray = xr.DataArray(TB, dims=["Time", "south_north", "west_east"])
 51
 52    TB_xarray = TB_xarray.chunk(rams_xarray["TOA_OLR"].chunksizes)
 53
 54    return TB_xarray

 57def rams_calculate_precip_rate(
 58    rams_xarray: xr.Dataset, calculation_type: str | None = None
 59) -> xr.DataArray:
 60    """
 61
 62
 63    Parameters
 64    ----------
 65    rams_xarray : xarray.core.dataset.Dataset
 66        Xarray Dataset containing default RAMS values.
 67    calculation_type : str
 68        A string dictating which variables to use for the precipitation rate calculation
 69
 70    Raises
 71    ------
 72        Exception if calculating precipitation rate with accumulation variables without ACCPR
 73        Exception if calculating precipitation rate with instantaneous rate variables without PCPRR
 74        Exception if calculating precipitation rate with volumetric instantaneous rate variables without PCPVR
 75        Exception if the calculation type is unknown
 76
 77    Returns
 78    -------
 79    total2D_precip_rate_xarray : xarray.core.dataarray.DataArray
 80        Xarray DataArray of precipitation rate in mm/hr.
 81
 82    """
 83
 84    # If calculation type is none, look at the available variables in rams xarray and determine which scheme to use
 85    if calculation_type is None:
 86        if "ACCPR" in rams_xarray:
 87            calculation_type = "surface time averaged precipitation rate"
 88
 89        elif "PCPRR" in rams_xarray:
 90            calculation_type = "surface instantaneous precipitation rate"
 91
 92        elif "PCPVR" in rams_xarray:
 93            calculation_type = "volumetric instantaneous precipitation rate"
 94
 95        else:
 96            raise Exception("No precipitation variables found in data")
 97
 98    # based on the calculation type, calculate precipitation rate
 99    if calculation_type == "surface time averaged precipitation rate":
100        accumulated_rainfall_vars = [
101            "ACCPR",
102            "ACCPP",
103            "ACCPS",
104            "ACCPA",
105            "ACCPG",
106            "ACCPH",
107            "ACCPD",
108            "ACONPR",
109            "ACCPIP",
110            "ACCPIC",
111            "ACCPID",
112        ]
113
114        if "ACCPR" not in rams_xarray:
115            raise Exception(
116                "If calculating precipitation with accumulation variables, ACCPR must be present"
117            )
118
119        total_accumulation = rams_xarray["ACCPR"]
120        for avar in accumulated_rainfall_vars[1:]:
121            if avar in rams_xarray:
122                total_accumulation += rams_xarray[avar]
123
124        pr = np.zeros_like(total_accumulation)
125
126        for tt in tqdm(
127            range(pr.shape[0] - 1),
128            desc="=====Calculating RAMS Precipitation Rate=====",
129            total=pr.shape[0] - 1,
130        ):
131            pr[tt] = (
132                total_accumulation[tt] - total_accumulation[tt - 1]
133            ) / rams_xarray.DT  # in kg / m^2 / s
134
135        # pr = pr / 1000 * 1000 # convert to mm / s, (kg / m^2 / s) * (1/1000 m^3/kg) * (1000 mm / m)
136        pr = xr.DataArray(pr, dims=["Time", "south_north", "west_east"]).chunk(
137            rams_xarray["TOPT"].chunksizes
138        )
139
140    elif calculation_type == "surface instantaneous precipitation rate":
141        instantaneous_rainfall_vars = [
142            "PCPRR",
143            "PCPRP",
144            "PCPRS",
145            "PCPRA",
146            "PCPRG",
147            "PCPRH",
148            "PCPRD",
149            "CONPRR",
150            "PCPRIP",
151            "PCPRIC",
152            "PCPRID",
153        ]
154
155        if "PCPRR" not in rams_xarray:
156            raise Exception(
157                "If calculating precipitation with instantaneous variables, PCPRR must be present"
158            )
159
160        pr = rams_xarray["PCPRR"]
161        for avar in instantaneous_rainfall_vars[1:]:
162            if avar in rams_xarray:
163                pr += rams_xarray[avar]
164
165        pr = pr.chunk(rams_xarray["TOPT"].chunksizes)
166
167    elif calculation_type == "volumetric instantaneous precipitation rate":
168        instantaneous_rainfall_vars = [
169            "PCPVR",
170            "PCPVP",
171            "PCPVS",
172            "PCPVA",
173            "PCPVG",
174            "PCPVH",
175            "PCPVD",
176            "CONPRR",
177            "PCPVIP",
178            "PCPVIC",
179            "PCPVID",
180        ]
181
182        if "PCPVR" not in rams_xarray:
183            raise Exception(
184                "If calculating precipitation with the volumetric instantaneous variables, PCPVR must be present"
185            )
186
187        pr = rams_xarray["PCPVR"]
188        for avar in instantaneous_rainfall_vars[1:]:
189            if avar in rams_xarray:
190                pr += rams_xarray[avar]
191
192        pr = pr.chunk(rams_xarray["PCPVR"].chunksizes)
193
194    else:
195        raise Exception("Unknown calculation type for precipitation rate calculation")
196    return pr

229def rams_calculate_reflectivity(rams_xarray: xr.Dataset) -> xr.DataArray:
230    """
231
232
233    Parameters
234    ----------
235    rams_xarray : xarray.core.dataset.Dataset
236        Xarray Dataset containing default RAMS values.
237
238    Returns
239    -------
240    dBZ: xarray.core.dataarray.DataArray
241        Xarray DataArray of calculated reflectivity values.
242
243    """
244
245    # constants defined by RAMS
246    p00 = 1.0e5  # Pa
247    rgas = 287  # J/gk/K
248    cp = 1004  # J/kg/K
249    cpor = cp / rgas
250
251    # find temperature, pressure, and grid point density
252    T_pot = rams_xarray["THETA"]  # potential temperature
253    PI = rams_xarray["PI"]
254    P = ((PI / cp) ** cpor) * p00 * 0.01  # convert pressure to mb
255    T = T_pot * PI / cp  # temperature in K
256    dens = (P * 100) / (T * rgas)
257
258    # mass coefficients defined by RAMS
259    alpha_mr = 524.0  # rain mass coeff
260    alpha_mg = 157.0  # graupel mass coeff
261    alpha_mh = 471.0  # hail mass coeff
262    alpha_mp = 110.8  # pris mass coeff
263    alpha_ms = 2.739e-3  # snow mass coeff
264    alpha_ma = 0.496  # aggregates mass coeff
265
266    # need mixing ratios and concentrations and ensure they are above 0
267    rainMix = xr.where(rams_xarray["RRP"] < 0, 0, rams_xarray["RRP"])
268    pris_iceMix = xr.where(rams_xarray["RPP"] < 0, 0, rams_xarray["RPP"])
269    snowMix = xr.where(rams_xarray["RSP"] < 0, 0, rams_xarray["RSP"])
270    aggregatesMix = xr.where(rams_xarray["RAP"] < 0, 0, rams_xarray["RAP"])
271    graupelMix = xr.where(rams_xarray["RGP"] < 0, 0, rams_xarray["RGP"])
272    hailMix = xr.where(rams_xarray["RHP"] < 0, 0, rams_xarray["RHP"])
273    rainConc = xr.where(rams_xarray["CRP"] < 0, 0, rams_xarray["CRP"])
274    pris_iceConc = xr.where(rams_xarray["CPP"] < 0, 0, rams_xarray["CPP"])
275    snowConc = xr.where(rams_xarray["CSP"] < 0, 0, rams_xarray["CSP"])
276    aggregatesConc = xr.where(rams_xarray["CAP"] < 0, 0, rams_xarray["CAP"])
277    graupelConc = xr.where(rams_xarray["CGP"] < 0, 0, rams_xarray["CGP"])
278    hailConc = xr.where(rams_xarray["CHP"] < 0, 0, rams_xarray["CHP"])
279
280    # define the gamma shape parameters
281    gamma = rams_xarray.gnu  # in order: cld rain pris snow aggr graup hail driz
282
283    dicToIterateThrough = {
284        "rain": [alpha_mr, rainMix, rainConc, gamma[1]],
285        "graupel": [alpha_mg, graupelMix, graupelConc, gamma[5]],
286        "pris_ice": [alpha_mp, pris_iceMix, pris_iceConc, gamma[2], 2.91],
287        "snow": [alpha_ms, snowMix, snowConc, gamma[3], 1.74],
288        "hail": [alpha_mh, hailMix, hailConc, gamma[6]],
289        "aggregates": [alpha_ma, aggregatesMix, aggregatesConc, gamma[4], 2.4],
290    }
291
292    # define an array of the reflectivity and iteratively add to it
293    dbz_total = np.zeros_like(rainMix)
294    Z_total = np.zeros_like(rainMix)
295
296    # thresholds for the mixing ratios and concentration numbers
297    q = 1e-10  # kg / kg
298    qn = 1e-3  # (# / kg)
299
300    for key in tqdm(
301        dicToIterateThrough.keys(),
302        desc="=====Calculating RAMS Reflectivity=====",
303        total=len(dicToIterateThrough.keys()),
304    ):
305        alpha = dicToIterateThrough[key][0]
306        mixRatio = dicToIterateThrough[key][1]
307        conc = dicToIterateThrough[key][2]
308        gsp = dicToIterateThrough[key][3]
309
310        F_gnu1 = (5.0 + gsp) * (4.0 + gsp) * (3.0 + gsp)
311        F_gnu2 = (2.0 + gsp) * (1.0 + gsp) * gsp
312        F_gnu = F_gnu1 / F_gnu2
313
314        # if the mixing ratio or number concentration is below a threshold there is no contribution to reflectivity
315        mixRatiobool = mixRatio > q
316        concbool = conc > qn
317        FullBool = mixRatiobool * concbool
318
319        if key in ["pris_ice", "snow", "aggregates"]:
320            factor = dicToIterateThrough[key][4]
321            M = mixRatio / conc
322            D = (M / alpha) ** (1 / factor)
323            alpha = M / (D**3)
324        tmp0 = mixRatio / alpha
325        tmp1 = (tmp0**2) * dens * F_gnu
326        tmp2 = (tmp1 / conc) * 1.0e18
327        tmp2 = np.where(FullBool, tmp2, 0)
328
329        Z_total += tmp2
330
331    Z_total = Z_total.clip(0.001, 1e99)
332
333    dbz_total = 10 * np.log10(Z_total)
334
335    dbz_total_xarray = xr.DataArray(
336        dbz_total, dims=["Time", "bottom_top", "south_north", "west_east"]
337    )
338
339    # Assign attributes
340    dBZ = dbz_total_xarray.assign_attrs(
341        {
342            "FieldType": 104,
343            "MemoryOrder": "XYZ",
344            "description": "radar reflectivity",
345            "units": "dBZ",
346            "stagger": "",
347            "coordinates": "Times GLON GLAT",
348        }
349    )
350
351    return dBZ.chunk(T.chunksizes)

282    gamma = rams_xarray.gnu  # in order: cld rain pris snow aggr graup hail driz
283
284    dicToIterateThrough = {
285        "rain": [alpha_mr, rainMix, rainConc, gamma[1]],
286        "graupel": [alpha_mg, graupelMix, graupelConc, gamma[5]],
287        "pris_ice": [alpha_mp, pris_iceMix, pris_iceConc, gamma[2], 2.91],
288        "snow": [alpha_ms, snowMix, snowConc, gamma[3], 1.74],
289        "hail": [alpha_mh, hailMix, hailConc, gamma[6]],
290        "aggregates": [alpha_ma, aggregatesMix, aggregatesConc, gamma[4], 2.4],
291    }
292
293    # define an array of the reflectivity and iteratively add to it
294    dbz_total = np.zeros_like(rainMix)
295    Z_total = np.zeros_like(rainMix)
296
297    # thresholds for the mixing ratios and concentration numbers
298    q = 1e-10  # kg / kg
299    qn = 1e-3  # (# / kg)
300
301    for key in tqdm(
302        dicToIterateThrough.keys(),
303        desc="=====Calculating RAMS Reflectivity=====",
304        total=len(dicToIterateThrough.keys()),
305    ):
306        alpha = dicToIterateThrough[key][0]
307        mixRatio = dicToIterateThrough[key][1]
308        conc = dicToIterateThrough[key][2]
309        gsp = dicToIterateThrough[key][3]
310
311        F_gnu1 = (5.0 + gsp) * (4.0 + gsp) * (3.0 + gsp)
312        F_gnu2 = (2.0 + gsp) * (1.0 + gsp) * gsp
313        F_gnu = F_gnu1 / F_gnu2
314
315        # if the mixing ratio or number concentration is below a threshold there is no contribution to reflectivity
316        mixRatiobool = mixRatio > q
317        concbool = conc > qn
318        FullBool = mixRatiobool * concbool
319
320        if key in ["pris_ice", "snow", "aggregates"]:
321            factor = dicToIterateThrough[key][4]
322            M = mixRatio / conc
323            D = (M / alpha) ** (1 / factor)
324            alpha = M / (D**3)
325        tmp0 = mixRatio / alpha
326        tmp1 = (tmp0**2) * dens * F_gnu
327        tmp2 = (tmp1 / conc) * 1.0e18
328        tmp2 = np.where(FullBool, tmp2, 0)
329
330        Z_total += tmp2
331
332    Z_total = Z_total.clip(0.001, 1e99)
333
334    dbz_total = 10 * np.log10(Z_total)
335
336    dbz_total_xarray = xr.DataArray(
337        dbz_total, dims=["Time", "bottom_top", "south_north", "west_east"]
338    )
339
340    # Assign attributes
341    dBZ = dbz_total_xarray.assign_attrs(
342        {
343            "FieldType": 104,
344            "MemoryOrder": "XYZ",
345            "description": "radar reflectivity",
346            "units": "dBZ",
347            "stagger": "",
348            "coordinates": "Times GLON GLAT",
349        }
350    )
351
352    return dBZ.chunk(T.chunksizes)

rams_xarray : xarray.core.dataset.Dataset Xarray Dataset containing default RAMS values.

Returns
